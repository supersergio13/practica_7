   }
  }

    printt ("a es diferente de b");
   } else {
    while ( (a!=0) && (b!=0) ) {
     a=a-1;
     b=b-1;
     if (a==0) {

       printt ("a es igual a b");

       printt ("a es diferente de b");
      }
     }
  }
  if (a!=0) {
   if (b==0) {
    printt ("a es diferente de b");
   } else {
    continue;
   }
  }
 }
